public static void mapAndSet(Collection<String> source, Map<String, String> map, Consumer<String> setter) {  
    if (!CollectionUtils.isEmpty(source) && map != null) {  
        String result = source.stream()  
                .map(map::get)  
                .filter(Objects::nonNull)  
                // 将每个值（如"01,11"）拆分为一个流（"01","11"），并进行扁平化处理  
                .flatMap(value -> Arrays.stream(value.split(",")))  
                .distinct()  
                .collect(Collectors.joining(","));  
        if (StringUtils.hasText(result)) {  
            setter.accept(result);  
        }  
    }  
}
